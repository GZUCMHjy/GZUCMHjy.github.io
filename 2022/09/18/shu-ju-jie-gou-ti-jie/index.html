1. 题目: 顺序表删除元素

1.1 Description ：已知线性表La存放数值，现需要删除指定下标为i的元素，并把删除元素后的La输出

1.2 Input：测试数据占两行，第一行为线性表La的元素，元素间以空格隔开，元素数量不超过100个。第二行为需要删除的小标i

1.3 Output: 输出删除指定下标为i的顺序表

  • 如图所示

  • tip：1.算法无难点，删除算法就是将后一项的元素逐一提前即可
    1. 难点在于对于顺序表的结构定义(废话不多说，开始上代码！！！)
      #include<stdio.h>
#define MaxSize 100//默认顺序表内存大小为100
#include<stdlib.h>
typedef int ElemType;
/*定义线性表*/
typedef struct
{
    ElemType list[MaxSize];//顺序表中用于存储的成员——数组list
    int last;            //用于纪录当前顺序表的长度的成员last
}SeqList;
/*创建空表*/
SeqList InitList()
{
    SeqList l;
    l.last=0;//初始化顺序表为空表,即元素个数为零,长度为零
    return l;
}
/*输入递增有序顺序表*/
int ListInsert(SeqList *L,int i,ElemType x) //i指的是要插入的元素的位置
{
    int j;
    if(L->last>=MaxSize)
    {
        printf("顺序表已经无法插入！\n");
        return 0;
    }
    else if(i<0||i>L->last)
    {
        printf("参数i不合法！\n");
        return 0;
    }
    else
    {
    for(j=L->last;j>=i;j--){
        L->list[j]=L->list[j-1];
    }
        L->list[i-1]=x;//空出来的位置插入目标数字x
        L->last++;//数组长度加一
        return 0;
    }
}

int ListaDelete(SeqList *L,int i)
{
    int j;
    if(L->last<=0)
    {
        printf("顺序表已经无空数据可删！\n");
        return 0;
    }
    else if(i<0||i>L->last-1)
    {
    //printf("参数i不合法");
        return -1;
    }
    else
    {
    //*x=L->list[i];
    for(j=i;j<=L->last;j++)
    {
        L->list[j]=L->list[j+1];//删除算法的核心就在这里！！！
    }
    L->last--;
    return 0;
    }
}

//输出元素
int OutputSeqList(SeqList *L)
{
    int i;
    //printf("输出结果为：");
    for(i=0;i<L->last;i++)
    {
        printf("%d ",L->list[i]);
    //return(L->list[i]);
    }
    return 0;
}
int main()
{
    SeqList l;
    l=InitList();//初始化空顺序表
    int pos;
    int i;
    for(i=0;i<5;i++){
        scanf("%d",&l.list[i]);
        l.last++;
    }
    scanf("%d",&pos);
    //ListInsert(&l,pos);
    ListaDelete(&l,pos);
    OutputSeqList(&l);
}

感悟: 抛开此题,应熟练书写顺序表的结构，并掌握crud的操作,即增删查改！给大家一个参考，我的顺序表结构的书写.

    #include<stdio.h>
#include<malloc.h>
#define Size 5
struct List//线性表List
{
    int *elem;//数组;
    int length;//元素个数;
    int size;
};
typedef struct List SqList;
SqList  IniList()//创建并初始化顺序表;
{
    SqList l;
    l.elem=(int*)malloc(Size*sizeof(int));//申请连续的动态内存空间;
    l.length=0;//初始化:元素个数为零;
    l.size=Size;//初始化:内存空间为Size;
    return l;//创建完毕,回到t;
}
int printList(SqList list)//遍历顺序表;
{
    int i;
    //printf("%d ",list.length);
    for(i=0;i<list.length;i++)
        printf("%d ",list.elem[i]);
        printf("\n");
        return 0;
}
int insert(SqList *t1,int elem,int pos)//特定位置插入数据;
{
    int *old_data=t1->elem;
    int i;
    if(pos<1||pos>t1->length)return -1;//printf("error.");//return false;//
    if(t1->length>=t1->size)//"==也可以"
    {
        t1->size*=2;;//printf("")
        t1->elem=(int *)malloc((t1->size)*sizeof(int));//数组(类型:int)申请内存空间扩大;
        //相应的size也要扩大;不一定加一;
        for(i=0;i<t1->length;i++)
            t1->elem[i]=old_data[i];
    }
  for(i=t1->length;i>=pos;i--)//pos决定于for要进行几次循环;length-pos=循环次数;
    {
        t1->elem[i]=t1->elem[i-1];
    }
    t1->elem[pos-1]=elem;
    t1->length++;//插入元素,自增;
    //printf("%d ",t1->length);
    //printList(&t1);
    return 0;
}
void delete (SqList *list,int pos)//指定位置删除;
{
    SqList *t1;
    t1=list;
    int i;
    for(i=pos-1;i<=t1->length;i++)
    {
        t1->elem[i]=t1->elem[i+1];
    }
    t1->length--;//删除元素,自减;
    //printList(&t1);
}
int main()
{
    int i;
    SqList la=IniList();
    //SqList *lb=IniList();
    for(i=0;i<5;i++)
    {
        scanf("%d",&la.elem[i]);
        la.length++;
    }
    printList(la);//printList(lb);
    insert(&la,95,3);
    printList(la);
    insert(&la,95,4);
    printList(la);
    insert(&la,95,5);
    printList(la);
    //insert(&lb,99,2);
    delete(&la,1);
    printList(la);
    delete(&la,3);
    printList(la);
    return 0;
}
