表单制作

|字数总计:432|阅读时长:2分钟|阅读量:

制作一个表单页面

<html>
    <head>
        <title>表单</title>
    </head>
    <body>
        

        <form>
            <h1 align="center">登录首页</h1>
            <input type="hidden" name="action" value="login"/>
            <table align="center">
                <tr>
                    <td>
                        用户名称:
                    </td>
                    <td>
                        <input name="username"type="text" value="默认值"/>
                    </td>
                </tr>
                <tr>
                    <td>
                        用户密码:
                    </td>
                    <td>
                         <input name="password"type="password" value="abc"/>
                    </td>
                </tr>
                <tr>
                    <td>
                        确认密码:
                    </td>
                    <td>
                         <input name="password"type="password" value="abc"/>
                    </td>
                </tr>
                <tr>
                    <td>
                        性别:
                    </td>
                    <td>
                         <input type="radio" name="sex" checked="checked"/>男<input type="radio"name="sex"/>女<br/>
                    </td>
                </tr>
                <tr>
                    <td>
                        兴趣爱好:
                    </td>
                    <td>
                        <input name="hobby"type="checkbox"checked="check"/>唱歌
                        <input name="hobby" type="checkbox"/>跳舞
                        <input name="hobby"type="checkbox"/>rap
                        <input name="hobby" type="checkbox"/>篮球
                    </td>
                </tr>
                <tr>
                    <td>
                        国籍:
                    </td>
                    <td>
                        <select name="country">
                        <option value="none">--请选择国籍</option>
                        <option selected="selected" value="cn">中国</option>
                        <option value="usa">美佬</option>
                        <option value="jap">小日本</option>
                        </select>
                    </td>
                </tr>
                <tr>
                    <td>
                        自我评价:
                    </td>
                    <td>
                        <textarea rows="10" cols="20">请输入自我评价信息</textarea>
                    </td>
                </tr>
                <tr>
                    <td>
                        <input type="reset" value="重置">
                        <input type="submit" value="提交">
                    </td>
                </tr>
                <!--action=login
                &username=默认值
                &password=123456
                &password=123456
                &sex=on
                &hobby=on
                &country=中国
                -->

            </table>
        </form>
    </body>
</html>
