发表于2022-08-18|更新于2022-11-14|学习记录
|字数总计:1.2k|阅读时长:5分钟|阅读量:
<html>
    <head>
        
        <meta charset="utf-8"/>
        <title>基本标签</title>
    </head>
    <body background="图片存放在本地路径"bgcolor="white"><!--bgcolor默认值为white-->
        <!--图片标签-->
        <!--src值为路径 alt表示报错返回值-->
        <image src="图片在本地的路径" alt="图片未找到"></image>
        <h1>
            标题1
        </h1><!--color默认值为黑色(black)-->
        <h2 color="green">标题2</h2>
        <br><!--换行-->
        <hr color="red"><!--水平线，独目标签(起到分隔的作用)-->
        <p>
            远离喧嚣，沉淀自己，成为更好的自己！<!--分段标签，另起一段-->
        </p>
        <!--字体展示效果(1)-->
        <b>粗体效果</b>
        <i>斜体效果</i>
        <!--字体展示效果(2)-->
        <!--运用字体标签font-->
        <font color ="red" size="12" >
            我喜欢你，粥粥！
        </font>
        <!--注释标签-->
        <!-- 表格标签-->
        <!--表格居中-->
        <table align="center" width="50%" height="200px" border="1px">
            <!--属性介绍-->
            <!--
				1px 表示一个像素
				border表示设置边框的宽度
			-->
            <!--第一行-->
            <tr align="center">
                <td>第一个单元格</td><!--第一列-->
                <td>第二个单元格</td><!--第二列-->
                <td align="center">第三个单元格</td><!--第三列-->
            </tr>
                <!--第二行-->
                <tr>
                    <!--标签<th>可以代替<td>-->
                    <!--额外的效果可以是单元格内容字体自动加粗和居中-->
                    <th>员工编号</th>
                    <th>员工姓名</th>
                    <th>员工性别</th>
            	</tr>
            <!--热链接-->
            <!--
				href值为跳转其他页面的网址链接
			-->
            <a href="htttp://www.jd.com">
                京东商城
            </a>
        </table>
        <!--列表标签-->
    <li>北京
        <!--<ul>标签的呈现样式跟<li>标签是一样的-->
        <ul type="disc">
            <!--<li>标签呈现形式是一个小实心的黑点-->
            <li>东城区</li>
            <li>西城区</li>
            <li>朝阳区</li>
            <li>海淀区</li>
        </ul>
    </li>
    <li>上海</li>
    <li>梅州</li>
    <li>南京</li>
    <li>新余</li>
        <li>
            茶
            <!--罗马数字I标识顺序-->
            <ol type="I">
                <li>绿茶
                    <!--阿拉伯数字标识顺序-->
                    <ol type="1">
                        <li>西红柿茶</li>
                        <li>龙井</li>
                        <li>龙尖</li>

                    </ol>
                </li>
                <li>红茶</li>

            </ol>
        </li>
    </body>
</html>

<html>
    <meta charset="utf-8"/>
    <head>
        <title>hehe</title>
    </head>
    <body>
        <form>
            <!--
				表单作用(form标签)
				1. 需要用户进行填写数据才可以向服务器发送请求
				2. 与超链接的区别在于 超链接只需要用户点击就可以向服务器发送请求
			-->
      
        用户名
        <!--
			输入框input
			
		-->
        <input type="text" name="username"/>
        密码
        <input type="password" name="userpwd"/>
        性别
        <!--
			checked 表示默认值
			radio 表示 圆点选项框
		-->
        <input type="radio" name="sex"value="1" checked>男
         <input type="radio" name="sex" value="0"checked>女
        兴趣
        <!--下拉列表多选-->
        <!--
			size 表示最多选几个
			mutiple 表示多选
		-->
        	<select name="兴趣"size="3" mutiple>
                <option value="c++" >
                    <!--
						类似map
						name      =     value
					-->
                   c++
                </option>
                <option value="java">
                   java
                </option>
                <option  value="JavaScript">
                    <!--相当于 兴趣=JavaScript (一一对应)-->
                   JavaScript
                </option>
       		 </select>
        <!--文本域-->
        <text row="20" col="20"></text>
        <!--checkbox 展现形式是方点选项框-->
        兴趣
            <input type="checkbox" name="兴趣" value="c++"/>c++
            <input type="checkbox" name="兴趣" value="c++"/>java
            <input type="checkbox" name="兴趣" value="c++"/>javascript
            <input type="submit"value="提交"/>
            <input type="reset"value="重置"/>
          </form>
    </body>
</html>
